const chai = require('chai');
const { transformPersonData } = require('./app'); // Import the function to test

const expect = chai.expect;

/**
 * @namespace UnitTests
 */

/**
 * Unit tests for the application functions.
 * @memberof UnitTests
 * @namespace transformPersonData
 */
describe('Unit Tests', () => {
  // Test transformPersonData function
  describe('transformPersonData', () => {
    /**
     * Test case for the transformPersonData function.
     * It should transform SWAPI data into the desired format.
     * @memberof UnitTests.transformPersonData
     * @function
     */
    it('should transform SWAPI data into the desired format', async () => {
      // Mock SWAPI response data
      const swapiData = {
        url: 'https://swapi.dev/api/people/1/',
        name: 'Luke Skywalker',
        birth_year: '19 BBY',
        gender: 'Male',
        height: '172',
        mass: '77',
        films: ['https://swapi.dev/api/films/1/'],
      };

      /**
       * Transform SWAPI data into the desired format.
       * @memberof UnitTests.transformPersonData
       * @function
       * @async
       * @param {Object} swapiData - SWAPI data to transform
       * @returns {Promise<Object>} - Transformed data
       */
      const transformedData = await transformPersonData(swapiData);

      // Define the expected transformed data
      const expectedTransformedData = {
        id: '1',
        name: 'Luke Skywalker',
        birth_year: '19 BBY',
        gender: 'Male',
        height: '172',
        weight: '77',
        movies: [
          {
            title: 'A New Hope',
            episode: 4,
            director: 'George Lucas',
            release_date: '1977-05-25',
          },
        ],
      };

      // Assert that the transformed data matches the expected data
      expect(transformedData).to.deep.equal(expectedTransformedData);
    });
  });
});
